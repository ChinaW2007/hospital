#### **步骤3: 启动时校验**

```typescript
export function verifyIntegrity(): boolean {
  // 1. 检查校验清单文件是否存在
  if (!fs.existsSync(INTEGRITY_PATH)) {
    console.error('❌ 校验清单 server/.integrity 不存在！');
    return false;
  }

  // 2. 解密校验清单
  const encrypted = fs.readFileSync(INTEGRITY_PATH);
  const json = decrypt(encrypted);  // AES-256-CBC解密
  const manifest = JSON.parse(json);

  // 3. 逐文件核验
  for (const [relPath, expectedHash] of manifest.files) {
    const absPath = path.join(ROOT, relPath);
    
    // 3.1 检查文件是否存在
    if (!fs.existsSync(absPath)) {
      console.error(`❌ 文件缺失: ${relPath}`);
      return false;
    }
    
    // 3.2 计算当前文件哈希（归一化换行符）
    const content = fs.readFileSync(filePath, 'utf8').replace(/\r\n/g, '\n');
    const actualHash = crypto.createHash('sha256').update(content).digest('hex');
    
    // 3.3 校验哈希是否匹配
    if (actualHash !== expectedHash) {
      console.error(`❌ 文件被篡改: ${relPath}`);
      console.error(`   期望: ${expectedHash.substring(0, 24)}...`);
      console.error(`   实际: ${actualHash.substring(0, 24)}...`);
      return false;
    }
  }

  // 4. 所有文件校验通过
  console.log('✅ 完整性校验通过');
  return true;
}
```

#### **步骤4: 运行时监控**

```typescript

## 4. 截止日期保护机制

### 4.1 实现位置

### 4.2 多层分布式保护架构

```
第0层（启动前）───────────────────────────────
  verify.ts: 文件完整性校验
  ↓ 校验失败 → 拒绝启动

第1层（全局中间件）─────────────────────────────
  index.ts: 全局截止日期检查
  ↓ 过期 → 401 Unauthorized

第2层（数据库拦截）─────────────────────────────
  db.ts: 包装query方法
  ↓ 过期 → SQL查询被拒绝

第3层（JWT认证）───────────────────────────────
  auth.ts: Token验证
  ↓ 过期 → 401 Unauthorized

第4层（权限控制）───────────────────────────────
  auth.ts: requireRole()
  ↓ 无权限 → 403 Forbidden

第5层（Token破坏）──────────────────────────────
  auth.ts: generateToken()
  ↓ 过期 → Token expiresIn=1ms（立即失效）

第6层（隐私哈希）──────────────────────────────
  auditChain.ts: 加盐哈希
  ↓ 所有敏感数据加密存储
```

export const SYSTEM_DEADLINE = '2026-12-31T23:59:59';
export const SYSTEM_DEADLINE_MESSAGE = '系统已停止服务';

export function isPastDeadline(): boolean {
  return new Date() > new Date(SYSTEM_DEADLINE);
}
```

#### **多层保护函数（guard.ts）**

```typescript
// 截止日期以数字分量形式存储，避免文本搜索
const LIMIT_Y = 2026;
const LIMIT_M = 12;   // 1-based
const LIMIT_D = 31;

function _hasExpired(): boolean {
  const n = new Date();
  const cy = n.getFullYear();
  const cm = n.getMonth() + 1;
  const cd = n.getDate();
  return cy > LIMIT_Y
    || (cy === LIMIT_Y && cm > LIMIT_M)
    || (cy === LIMIT_Y && cm === LIMIT_M && cd > LIMIT_D);
}

// ── 供不同层级调用的出口，各自独立命名 ──

/** 全局 HTTP 中间件使用 */
export function rejectIfExpired(): boolean {
  return _hasExpired();
}

/** 数据库查询拦截使用 */
export function isLicenseInvalid(): boolean {
  return _hasExpired();
}

/** Token 生成 sabotage 使用 */
export function shouldSabotageToken(): boolean {
  return _hasExpired();
}
```

#### **数据库查询拦截**

```typescript
// db.ts
const _rawQuery = pool.query.bind(pool);

(pool as any).query = function () {
  if (isLicenseInvalid()) {
    return Promise.reject(new Error('系统已停止服务'));
  }
  return (_rawQuery as any).apply(pool, arguments);

```

#### **JWT验证**

```typescript
export function authMiddleware(req: Request, res: Response, next: NextFunction): void {
  // 1. 检查系统截止日期
  if (isPastDeadline()) {
    res.status(401).json({ error: SYSTEM_DEADLINE_MESSAGE });
    return;
  }

  // 2. 提取Token
  const authHeader = req.headers.authorization;
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    res.status(401).json({ error: '未登录，请先登录' });
    return;
  }

  const token = authHeader.substring(7);
  
  // 3. 验证Token
  try {
    const decoded = jwt.verify(token, JWT_SECRET) as AuthUser;
    req.user = decoded;  // 注入用户信息
    next();
  } catch {
    res.status(401).json({ error: '登录已过期，请重新登录' });
  }
}
```

### 5.3 权限控制

```typescript
export function requireRole(...roles: string[]) {
  return (req: Request, res: Response, next: NextFunction): void {
    if (!req.user) {
      res.status(401).json({ error: '未登录' });
      return;
    }
    
    if (!roles.includes(req.user.role)) {
      res.status(403).json({ error: '权限不足' });
      return;
    }
    
    next();
  };
}

// 使用示例
router.post('/', requireRole('doctor', 'admin'), async (req, res) => {
  // 只有医生和管理员可以创建处方
});
```

| **角色权限控制** | 基于角色的访问控制（RBAC） |
| **防止伪造** | 使用JWT_SECRET签名验证 |

const auditSalt = process.env.AUDIT_HASH_SALT || 'his-local-audit-salt';

const hashPrivateValue = (label: string, value: string | number | null) => {
  if (value === null || value === undefined || value === '') return null;
  return hash(`${auditSalt}:${label}:${String(value)}`);
}

// 示例: 加盐哈希敏感数据
